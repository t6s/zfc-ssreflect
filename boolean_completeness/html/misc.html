Require Import Eqdep_dec.

Lemma inj_right_pair2 :
 ∀ A : Set,
 (∀ x y : A, {x = y} + {x ≠ y}) →
 ∀ (x : A) (P : A → Set) (y y' : P x),
 existS P x y = existS P x y' → y = y'.
